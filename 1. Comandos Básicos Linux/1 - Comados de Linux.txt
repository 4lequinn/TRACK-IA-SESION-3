// Mover
mv <nombre_archivo> <ruta_destino>
(ejem) mv apuntes.txt cursos/programacion/

// Copiar
cp <nombre_archivo> <ruta_destino>

// Editar archivo y/o crear
nano <nombre_archivo> 
(ctrl + s = guardar)
(ctrl + x = salir)

// Crear directorios
mkdir <nombre_directorio>


// Ver contenido de un archivo
cat <nombre_archivo>

// Eliminar
rm <nombre_archivo>

// Eliminar de forma recursiva (forzada)
rm -rf <nombre_archivo>
